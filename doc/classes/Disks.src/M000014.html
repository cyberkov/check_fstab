                def check_for_changes
                        # Check on disk changes
                        unless @disks[0][:changes] == false
                                        $log.info "Cachefile changed by user. Honoring changes."
                                        @disks[0][:changes] = false
                                        write_disk_cache
                                        return true
                        end
                        return false
                end
