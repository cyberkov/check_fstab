                def read_disk_cache
                                if File.exists?("#{$cache_path}/#{@host}_disks.yaml")
                                                @disks = YAML::load(File.read("#{$cache_path}/#{@host}_disks.yaml"))
                                                $log.info "Loaded cachefile for #{@host}."
                                else
                                                $log.warn "No cachefile found. Need to get a new one."
                                                return false
                                end
                end
