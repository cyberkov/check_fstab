                def update_disks(disks)
                        unless @disks.nil?
                                @disks.each do |olddisk|
                                                found = false
                                                disks.each_with_index do |disk,i|
                                                                next unless disk[:disk] == olddisk[:disk]
                                                                disk[:warn] = olddisk[:warn]
                                                                disk[:crit] = olddisk[:crit]
                                                                found = true
                                                                break
                                                end
                                                next if found == true
                                                # hmm. disk has vanished. add it to new array.
                                                disks << olddisk
                                end

                        end
                        @disks = disks
                end
